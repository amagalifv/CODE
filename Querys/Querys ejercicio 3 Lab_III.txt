FROM CLIENTES C 
	INNER JOIN CUxCLI CC ON C.DNI_CLI=CC.DNI_CLI
WHERE C.DNI_CLI=23

SELECT C.DNI_CLI, C.NOMBRE_CLI, C.APELLIDO_CLI, C.DIRECCION_CLI, CC.ID_SUCURSAL_CUxCLI, S.NOMBRE_S
FROM CLIENTES C 
	INNER JOIN CUxCLI CC ON C.DNI_CLI=CC.DNI_CLI
	INNER JOIN CUxCLI CC ON CC.ID_SUCURSAL_CUxCLI=S.ID_SUCURSAL_S
WHERE C.DNI_CLI=23

SELECT MAX(T.IMPORTE), T.ID_TRANSACCION, T.ID_CUENTA, C.DNI_CLI, C.NOMBRE_CLI, C.APELLIDO_CLI, C.DIRECCION_CLI
FROM CLIENTES C 
	INNER JOIN CUxCLI CC ON C.DNI_CLI=CC.DNI_CLI
	INNER JOIN TRANSACCIONES T ON T.ID_CUENTA=CC.ID_CUENTA

select Monto_T as [Maxima Transaccion], Clientes.Nombre_Cli
from Transacciones inner join Clientes 
on Transacciones.DNI_Cliente_T = Clientes.DNI_Cli
where Monto_T = (select MAX(Monto_T) from transacciones)
GO
